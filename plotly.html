Instacart Plots - Plotly

library(tidyverse)
library(p8105.datasets)
library(plotly)
data("instacart")

instacart = 
  instacart %>% 
  select(-eval_set, -add_to_cart_order, -order_id, -days_since_prior_order)
instacart %>% 
    count(aisle) %>% 
    mutate(
        aisle = factor(aisle),
        aisle = fct_reorder(aisle, n)
    ) %>% 
    plot_ly(y = ~n, x = ~aisle, color = ~aisle,
            type = "bar", colors = "viridis")
